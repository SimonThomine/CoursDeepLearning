    def conv_bn_lrelu(in_channels, out_channels, kernel_size, stride, padding):
  return nn.Sequential(
    nn.Conv2d(in_channels, out_channels, kernel_size, stride, padding,bias=False),
    nn.BatchNorm2d(out_channels),
    nn.LeakyReLU()
  )

class discriminator(nn.Module):
  def __init__(self, features_d=64) -> None:
    super().__init__()
    self.discr = nn.Sequential(
      conv_bn_lrelu(1, features_d, kernel_size=3, stride=2, padding=1),
      conv_bn_lrelu(features_d, features_d*2, kernel_size=3, stride=2, padding=1),

      nn.Conv2d(256, 1, kernel_size=3, stride=2, padding=0),
      nn.Sigmoid()
    )

    return self.discr(x)

    Avant de consttuire notre boucle d'entra√Ænement, r√©capitulons les √©tapes que nous devons faire :

    • On commence par r√©cuperer batch_size √©l√©ments du dataset d'entra√Ænement et on pr√©dit les labels avec notre discriminateur
    • Ensuite, on g√©n√©re batch_size √©l√©ments avec notre g√©n√©rateur et on pr√©dit les labels
    • On va ensuite mettre √† jour les poids du mod√®le discriminateur √† partir des deux loss
    • On va ensuite √† nouveau pr√©dire les labels des donn√©es g√©n√©r√©es (car on a mis √† jour le discriminateur entre temps)
    • A partir de ces valeurs, on calcule le loss et on met √† jour notre g√©n√©rateur
